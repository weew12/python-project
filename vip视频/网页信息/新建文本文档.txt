	var s, vurl;

			vurl = s[0] + "|" + s[1].substring(0, s[1].length - 5) + s[1].substring(s[1].length - 4) + "|" + s[2] + "|";
		}
	} else if (from == "gvod") {
		s = url.replace("gvod://", "").split("/");
		vurl = "gvod://" + s[0] + "/" + s[1].substring(0, s[1].length - 5) + s[1].substring(s[1].length - 4) + "/" + s[2] + "/" + s[3];
